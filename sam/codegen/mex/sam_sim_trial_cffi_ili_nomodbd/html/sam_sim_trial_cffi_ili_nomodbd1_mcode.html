 11% SAM trial simulation function, modeling choice as feed-forward
 12% inhibition, inhibition as blocking the input, and excluding extrinisic
 13% (B) or intrinisic (D) modulation of connectivity.
 14%
 15% SYNTAX
 16% Let there be M inputs, N units, P time points, then
 17% u - inputs to the accumulators (MxP double)
 18% A - endogenous connectivity matrix (NxN double)
 19% B - extrinsic modulation matrix (NxNxM double)
 20% C - exogenous connectivity matrix (NxM double)

 37% m - number of inputs (1x1 double)

 64z(:,1) = Z0;
 65
 66% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 67% 1. STOCHASTIC ACCUMULATION PROCESS
 68% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 69
 70while t < p - 1

 72  % Endogenous connectivity at time t (note that A is a function of t
 73  % because lateral inhibition kicks in once a unit has reached its
 74  % threshold)
 75  % -----------------------------------------------------------------------
 76  At = A;
 77  
 78% % Extrinsic modulation at time t
 79% % -----------------------------------------------------------------------
 80% Bt = zeros(m,m);
 81% for i = 1:m
 82% Bt = Bt + u(i,t)*B(:,:,i);
 83% end
 84
 85% % Intrinsic modulation at time t
 86% % -----------------------------------------------------------------------
 87% Dt = zeros(n,n);
 88% for j = 1:n
 89% Dt = Dt + z(j,t)*D(:,:,j);
 90% end
 91

 94% u(any(blockInput(:,resp),2),t) = 0;

 97  % -----------------------------------------------------------------------
 98  At(latInhib(:,~resp)) = 0;
 99

101% % -----------------------------------------------------------------------
102% dzdt = (At + Bt + Dt) * z(:,t) * dt/tau + ... %
103% C * u(:,t) * dt/tau + ... % Inputs
104% SI * randn(n,1) * sqrt(dt/tau); % Noise (in)
105              
106  dzdt = At * z(:,t) * dt/tau + ... % Endogenous connectivity
107                C * u(:,t) * dt/tau + ... % Inputs
108                SI * randn(n,1) * sqrt(dt/tau); % Noise (in)
109              
110  % Log new activation level
111  % -----------------------------------------------------------------------
112  z(:,t+1) = z(:,t) + dzdt;
113
114  % Rectify activation if below zLB
115  % -----------------------------------------------------------------------
116  z(z(:,t+1) < ZLB,t+1) = ZLB(z(:,t+1) < ZLB);
117
118  % Identify units that crossed threshold
119  % -----------------------------------------------------------------------
120  resp(z(:,t+1) > ZC) = true;
121  
122  % Determine time of crossing threshold
123  % -----------------------------------------------------------------------
124  rt(resp & isinf(rt)) = T(t + 1);
125  
126  % Break accumulation if termination criterion has been met
127  % -----------------------------------------------------------------------
128  if any(terminate(resp))
129    break
130  end
131  
132  % Update time
133  % -----------------------------------------------------------------------
134  t = t + 1;
135
